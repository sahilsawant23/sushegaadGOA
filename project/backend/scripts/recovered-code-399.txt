const fs = require('fs');
const readline = require('readline');
const path = require('path');

const logPath = 'C:\\Users\\hp\\.gemini\\antigravity-ide\\brain\\1451c98c-fbc6-44be-8f42-253cd51bf054\\.system_generated\\logs\\transcript_full.jsonl';

async function search() {
  const fileStream = fs.createReadStream(logPath);
  const rl = readline.createInterface({
    input: fileStream,
    crlfDelay: Infinity
  });

  for await (const line of rl) {
    if (line.includes('realtime/places')) {
      console.log('--- FOUND MATCH ---');
      const data = JSON.parse(line);
      if (data.tool_calls) {
        data.tool_calls.forEach(tc => {
          if (tc.name === 'replace_file_content' || tc.name === 'write_to_file') {
            console.log('Tool Args:', JSON.stringify(tc.args, null, 2));
          }
        });
      } else if (data.content && data.content.includes('realtime/places')) {
        console.log('Content preview:', data.content.substring(0, 1500));
      }
    }
  }
}

search();
